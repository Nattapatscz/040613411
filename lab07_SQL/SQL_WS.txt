SELECT
    orders.username,
    orders.ord_date,
    SUM(product.price * item.quantity) AS total_amount
FROM
    orders
JOIN
    item ON orders.ord_id = item.ord_id
JOIN
    product ON item.pid = product.pid
GROUP BY
    orders.username,
    orders.ord_date
ORDER BY
    orders.ord_date;
